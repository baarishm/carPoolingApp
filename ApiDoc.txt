1. For signup

 URL : 127.0.0.1:8000/api/auth/signup
 Method : POST
 Headers : 
    Content-Type:application/json
    X-Requested-With:XMLHttpRequest
 Body : raw
    {
        "first_name":"api",
        "last_name":"Test",
        "email":"api@test.com",
        "phone":"9876543210",
        "password":"123456",
        "password_confirmation":"123456"
    }
 Response :
    {
        "status": true,
        "error": [],
        "message": "Successfully created user!"
    }

3. For Logout

 URL : 127.0.0.1:8000/api/auth/logout
 Method : POST
 Headers : 
    Content-Type:application/json
    X-Requested-With:XMLHttpRequest
 Body : raw
    {
        "phone":"9876543210",
        "accessToken": <token received when logged in>
    }
 Response :
    {
        "status": true,
        "error": [],
        "message": "Successfully logged out"
    }
